  def get_array_of_sentences
    result = Array.new
    @xml_document.elements.each(@@xpath_to_sentence) {
      |e|
      temp = ""
      e.elements.each(@@xpath_to_word) {
        |q|
        temp << q.text << "/" << q.attributes[@@attribute_type] << " "
        }
        result.push temp
    }
    return result
  end
